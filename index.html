class Node {

        this.data = data;
    }
}

public class ReverseLinkedList {

    public static void main(String[] args) {

        Node head = new Node(1);

        head.next = new Node(2);

        head.next.next = new Node(3);

        Node prev = null;

        Node current = head;

        while(current != null) {

            Node next = current.next;

            current.next = prev;

            prev = current;

            current = next;
        }

        while(prev != null) {

            System.out.println(prev.data);

            prev = prev.next;
        }
    }
}
16. Detect Loop in Linked List ▼ 
class Node {

        this.data = data;

}

public class DetectLoop {

    public static void main(String[] args) {

        Node head = new Node(1);

        head.next = new Node(2);

        head.next.next = new Node(3);

        head.next.next.next = head;

        Node slow = head;

        Node fast = head;

        boolean loop = false;

        while(fast != null && fast.next != null) {

            slow = slow.next;

            fast = fast.next.next;

            if(slow == fast) {

                loop = true;

                break;
            }
        }

        System.out.println(loop);
    }
}

1. Inheritance & Polymorphism ▼

This example demonstrates:

  • Inheritance

Real-Time Banking Architecture

Class Purpose
Account Base class
SavingsAccount Minimum balance validation
CurrentAccount Overdraft support 
class Account {

    protected String accountHolder;

    protected double balance;

    public Account(String accountHolder,
                   double balance) {

        this.accountHolder = accountHolder;

        this.balance = balance;
    }

    public void deposit(double amount) {

        balance += amount;

        System.out.println(
                amount + " deposited");
    }

    public void withdraw(double amount) {

        if(balance >= amount) {

            balance -= amount;

            System.out.println(
                    amount + " withdrawn");

        } else {

            System.out.println(
                    "Insufficient Balance");
        }
    }
}

class SavingsAccount extends Account {

    private double minimumBalance = 1000;

    public SavingsAccount(String accountHolder,
                          double balance) {

        super(accountHolder, balance);
    }

    @Override
    public void withdraw(double amount) {

        if((balance - amount)
                >= minimumBalance) {

            balance -= amount;

            System.out.println(
                    "Savings Withdrawal Success");

        } else {

            System.out.println(
                    "Minimum Balance Required");
        }
    }
}
Important Interview Point: "Parent reference + child object achieves runtime polymorphism."
